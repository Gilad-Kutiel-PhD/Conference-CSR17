\usetikzlibrary{
	positioning,
	plotmarks,
}

\tikzset{
	on/.code args={<#1>#2}{
  		\only<#1>{\pgfkeysalso{#2}}
	},
}

\tikzset{
label/.style={
	midway,
	sloped,
	draw=none,
	fill=white,
	inner sep=1pt,
	minimum height=0,
	minimum width=0,
},
label below/.style={label, below},
label above/.style={label, above},
}

\tikzset{blur/.style={black!10}}

\tikzset{default node/.style={
	draw, 
	circle,
	inner sep=0mm,
	minimum size=5mm,
	very thick,
	font=\small,
	black!70,
}}

\tikzset{
	hide nodes/.style={
		every node/.append style={opacity=0}
	},
	hide nodes text/.style={
		every node/.append style={text=white}
	},
	show nodes/.style={
		every node/.append style={opacity=1}
	},
	show nodes text/.style={
		every node/.append style={text=black}
	},
	nodes pass/.append style={
		every node/.append style={\colorPass}
	},
	nodes drive/.append style={
		every node/.append style={\colorDrive}
	},
}

\tikzset{
	>=latex,
	->,
	every node/.style={default node},
}
